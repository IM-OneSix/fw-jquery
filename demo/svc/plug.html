$svc.plug('', functoin($param, $return) {
  var tpl = '';
  // 뷰바인딩
  var vo=$svc.bind({name:'', plug:tpl});
  // 뷰이벤트핸들러
  var on=vo.event();
  // 뷰렌더링
  vo.render();
  // 모창으로 전달
  on.close=function(){
    $return();
  };
});
